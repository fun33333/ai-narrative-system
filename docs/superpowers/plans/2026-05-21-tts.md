```python
@app.get("/api/tts")
async def api_tts(text: str, speaker: str = ""):
    """Generate TTS audio for given text. Returns MP3 stream."""
    voice = SPEAKER_TO_VOICE.get(speaker, DEFAULT_VOICE)
    try:
        communicate = edge_tts.Communicate(text=text, voice=voice)
        buf = io.BytesIO()
        async for chunk in communicate.stream():
            if chunk["type"] == "audio":
                buf.write(chunk["data"])
        buf.seek(0)
        return StreamingResponse(buf, media_type="audio/mpeg")
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

- [ ] **Step 4: Add `speakTurn` function** (inside `Home()`, after `stopAudio`)

```javascript
const speakTurn = async (turn) => {
  if (!turn) return;
  stopAudio();
  const text = turn.dialogue || "";
  if (!text.trim()) return;
  const speaker = characterVoices[turn.character] || "";
  setIsTTSLoading(true);
  try {
    const res = await fetch(
      `${API_BASE}/api/tts?text=${encodeURIComponent(text)}&speaker=${encodeURIComponent(speaker)}`
    );
    if (!res.ok) { setIsTTSLoading(false); return; }
    const blob = await res.blob();
    const url = URL.createObjectURL(blob);
    const audio = new Audio(url);
    audioRef.current = audio;
    setIsTTSLoading(false);
    setIsSpeaking(true);
    audio.onended = () => { setIsSpeaking(false); audioRef.current = null; };
    audio.onerror = () => { setIsSpeaking(false); audioRef.current = null; };
    audio.play();
  } catch (_) {
    setIsTTSLoading(false);
  }
};
```
